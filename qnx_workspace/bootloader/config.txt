# Enable audio (loads snd_bcm2835)
dtparam=audio=on

# Additional overlays and parameters are documented
# /boot/firmware/overlays/README

# Automatically load overlays for detected cameras
camera_auto_detect=1

# Automatically load overlays for detected DSI displays
display_auto_detect=1

# Automatically load initramfs files, if found
auto_initramfs=1

# Enable DRM VC4 V3D driver
dtoverlay=vc4-kms-v3d
max_framebuffers=2

# Don't have the firmware create an initial video= setting in cmdline.txt.
# Use the kernel's default instead.
disable_fw_kms_setup=1

# Run in 64-bit mode
arm_64bit=1

# Enable PL011 UART for FTDI Serial Console
enable_uart=1
dtoverlay=uart3

# CRITICAL: Disable BT to route PL011 to GPIO 14/15 for FTDI Console but comment out if miniuart is used
#dtoverlay=disable-bt

# Lock the core frequency to prevent UART clock drift
core_freq=500
core_freq_min=500

# Force the UART base clock to exactly 48MHz to match the QNX .build file
init_uart_clock=48000000

# Enable early bootloader UART logging
uart_2ndstage=1

# Point the bootloader to the QNX IFS payload
kernel=ifs-rpi4.bin

# Disable compensation for displays with overscan
disable_overscan=1

# Run as fast as firmware / board allows
arm_boost=1

[cm4]
# Enable host mode on the 2711 built-in XHCI USB controller.
# This line should be removed if the legacy DWC2 controller is required
# (e.g. for USB device mode) or if USB support is not required.
otg_mode=1

[cm5]
dtoverlay=dwc2,dr_mode=host
